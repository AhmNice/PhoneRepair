search_icon.addEventListener('click', () => {
    if (elementTarget.value.trim() !== '') { // Check if the input value is not empty
        // Clear existing reports
        reportContainer.innerHTML = '';
       
        // Loop through titles
        titles.forEach(title => {
            if (title.textContent.toLowerCase().includes(elementTarget.value.toLowerCase())) {
                const report = document.createElement('div');
                report.className = "list";
                const text = document.createElement('h3');
                text.textContent = title.textContent;
                report.appendChild(text);
                reportContainer.appendChild(report);
            }
        });
    }
});
